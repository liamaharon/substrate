Expand description

Converts a type to an OutputResult.

This is for example implemented on std::process::Output.

Examples

use assert_cmd::prelude::*;

use std::process::Command;

let result = Command::new("echo")
    .args(&["42"])
    .ok();
assert!(result.is_ok());
